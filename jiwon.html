Starting Grammar

ex 2.3

#include <stdio.h>
int main(void)
{
  printf("Hello World!");
  return 0;
}

#include <stdio.h>

statement started with # is preprocessor indicator. Do not end this sentence with ;(b/c this is not ‘sentence’)

  • preprocessor: part of compiler. pre-stop before compile.
  • #include header file: command to include particular file to present location
  • header file: have information which compiler needs
  • stdio.h: header file which has information about input/ouput func.
  • almost programs use more than one header file
  • empty line don’t affects to result
int main(void)

statement which defines func’s output/input type, name. This defines func main()

  • int main(void): value’s type that func prints is integer
  • void: empty. func’s input is empty.
{
}

show where is func’s start and end. between them, put func’s body which peforms func’s function. put ; end of the statement.

printf("Hello World!")

call printf() func and print txt on the screen.

  • f: short for formatted
  • i.e print func which decides format
return 0;
  • return: use when func finishes the process and return the result
  • when execute sentence return 0;, main() func ends the process and put out the value to outside
  • outside: OS(Operation System) which runs this program

  • usally, value of 0 means the program ends normally

p.100~p.103

Program Examples

ex.1: Calculation Annual Salary Program

//저축액을 계산하는 프로그램
#include <stdio.h>

int main(void)
{
	int salary;	//월급
	int deposit; //저축액

	printf("월급을 입력하시오:");
	scanf("%d", &salary); //입력받은 값을 salary 변수 주소에 저장

	deposit = 10 * 12 * salary;

	printf("10년 동안의 저축액: %d\n", deposit);

	return 0;
}

ex.2: Calculation Circle’s Demension Program

//원의 면적을 계산하는 프로그램
#include <stdio.h>
int main(void)
{
	float radius;	//원의 반지름
	float area;		//면적

	printf("반지름을 입력하시오:");
	scanf("%f", &radius);

	area = 3.14 * radius * radius;

	printf("원의 면적: %f\n", area);

	return 0;
}

ex.3: Calculation Exchange Rate Program

//환율을 계산하는 프로그램
#include <stdio.h>

int main(void)
{
	float rate;	//원/달러 환율
	float usd;	//달러화
	int krw;	//원화

	printf("달러에 대한 원화 환율을 입력하시오:");	//입력 안내 메시지
	scanf("%f", &rate);								//사용자로부터 환율 입력

	printf("원화 금액을 입력하시오:");				//입력 안내 메시지
	scanf("%d", &krw);								//원화 금액 입력

	usd = krw / rate;								//달러화로 환산

	printf("원화 %d원은 %f달러입니다.\n", krw, usd);	//계산 결과 출력

	return 0;
}

ex.4: Calculation Square’s perimeter and dimension Program

#include <stdio.h>

int main(void)
{
	double w,h,area,perimeter;

	printf("직사각형의 가로 길이를 입력하세요:");
	scanf("%lf",&w);

	printf("직사각형의 세로 길이를 입력하세요:");
	scanf("%lf",&h);

	area = w * h;
	perimeter = 2 * (w + h);

	printf("직사각형의 둘레는 %1f이고 면적은 %1f입니다.\n", area, perimeter);

	return 0;
}
